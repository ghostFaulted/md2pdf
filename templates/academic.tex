% Custom environment for Obsidian Callouts
% Argument 1: Type (note, warning, danger, etc.) - mapped to <type>color
% Argument 2: Title of the callout
\newtcolorbox{obscallout}[2]{
    enhanced,
    breakable,
    frame hidden,
    boxrule=0pt,
    % Left border line: 4pt width, 0pt offset, color based on type
    borderline west={4pt}{0pt}{#1color},
    % Background color: 5% opacity of the main color
    colback=#1color!5!white,
    % Title styling
    coltitle=#1color,
    fonttitle=\bfseries,
    title={#2},
    % Layout adjustments
    attach title to upper=\par\vspace{0.3em},
    sharp corners,
    left=8pt,
    right=8pt,
    top=8pt,
    bottom=8pt,
    before skip=1em,
    after skip=1em
}
